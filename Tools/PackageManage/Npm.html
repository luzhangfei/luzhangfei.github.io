# NPM常用命令

# npm install 安装模块

npm install (with no args,in package dir)
npm install <name>@<tag>
npm install <name>@<version>
npm install <name> --save/-S   加入dependencies生产阶段依赖
npm install <name> --save-dev/-D   加入devDependencies开发阶段依赖

# npm uninstall 卸载模块

npm uninstall <name>[@<version>] [-S|--save|-D|--save-dev]

# npm update 更新模块

npm update [-g] [<pkg>]

# npm outdated 检查模块是否已经过时

npm outdated [<pkg>]

# 查看已经安装的模块

npm ls [<pkg>] [-g]

# 创建新项目

npm init

# npm root查看包的安装路径

npm root [-g]

# npm cache 管理模块的缓存

npm cache add <name>@<version>
npm cache ls [<path>]
npm cache clean [<path>]
